270  const Quaternion_t<T> Integrate(const vec3_t<T>& angularVelocity, T deltaTime) const
271  {
272  Quaternion_t<T> deltaQ(DO_NOT_INITIALIZE);
273  auto scaledAngularVelocity = angularVelocity * (deltaTime * T(0.5));
274  auto squaredVelocityLength = scaledAngularVelocity.squaredLength();
275 
276  T s = 1;
277 
278  if(squaredVelocityLength * squaredVelocityLength / T(24) < GetEpsilon<T>::value())
279  {
280  deltaQ.angle = T(1) - squaredVelocityLength / T(2);
281  s = T(1) - squaredVelocityLength / T(6);
282  }
283  else
284  {
285  auto velocityLength = gep::sqrt(squaredVelocityLength);
286 
287  deltaQ.angle = gep::cos(velocityLength);
288  s = gep::sin(velocityLength) / velocityLength;
289  }
290 
291  deltaQ.x = scaledAngularVelocity.x * s;
292  deltaQ.y = scaledAngularVelocity.y * s;
293  deltaQ.z = scaledAngularVelocity.z * s;
294 
295  return deltaQ * (*this);
296  }
