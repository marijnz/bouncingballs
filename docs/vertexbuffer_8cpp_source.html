169 void gep::Vertexbuffer::upload(ID3D11DeviceContext* pDeviceContext)
170 {
171  if(m_isDataUploaded && m_uploadedData < m_data.length())
172  {
173  m_pDataBuffer->Release();
174  m_pDataBuffer = nullptr;
175  m_isDataUploaded = false;
176  }
177  if(m_data.length() > 0)
178  {
179  if(!m_isDataUploaded)
180  {
181  m_dataDesc.Usage = (m_usage == Usage::Dynamic) ? D3D11_USAGE_DYNAMIC : D3D11_USAGE_IMMUTABLE;
182  m_dataDesc.ByteWidth = (UINT)(m_data.length() * sizeof(float));
183  m_dataDesc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
184  m_dataDesc.CPUAccessFlags = (m_usage == Usage::Dynamic) ? D3D11_CPU_ACCESS_WRITE : 0;
185  m_dataDesc.MiscFlags = 0;
186  m_dataDesc.StructureByteStride = 0;
187 
188  HRESULT hr = S_OK;
189  D3D11_SUBRESOURCE_DATA initData;
190  initData.pSysMem = m_data.toArray().getPtr();
191  initData.SysMemPitch = 0;
192  initData.SysMemSlicePitch = 0;
193 
194  hr = m_pDevice->CreateBuffer(&m_dataDesc, &initData, &m_pDataBuffer);
195  if(FAILED(hr))
196  throw Exception("Failed to create vertex data buffer");
197 
198  m_isDataUploaded = true;
199  m_uploadedData = m_data.length();
200  }
201  else
202  {
203  D3D11_MAPPED_SUBRESOURCE res;
204  HRESULT hr = pDeviceContext->Map(m_pDataBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &res);
205  GEP_ASSERT(SUCCEEDED(hr));
206  auto newData = m_data.toArray();
207  memcpy(res.pData, newData.getPtr(), newData.length() * sizeof(float));
208  pDeviceContext->Unmap(m_pDataBuffer, 0);
209  }
210  }
211 
212  if(m_areIndicesUploaded && m_uploadedIndices < m_indices.length())
213  {
214  m_pIndexBuffer->Release();
215  m_pIndexBuffer = nullptr;
216  m_areIndicesUploaded = false;
217  }
218  if(m_indices.length() > 0)
219  {
220  m_hasIndexBuffer = true;
221  if(!m_areIndicesUploaded)
222  {
223  m_indexDesc.Usage = (m_usage == Usage::Dynamic) ? D3D11_USAGE_DYNAMIC : D3D11_USAGE_IMMUTABLE;
224  m_indexDesc.ByteWidth = (UINT)(m_indices.length() * sizeof( uint32 ));
225  m_indexDesc.BindFlags = D3D11_BIND_INDEX_BUFFER;
226  m_indexDesc.CPUAccessFlags = (m_usage == Usage::Dynamic) ? D3D11_CPU_ACCESS_WRITE : 0;
227  m_indexDesc.MiscFlags = 0;
228 
229  HRESULT hr = S_OK;
230  D3D11_SUBRESOURCE_DATA initData;
231  initData.pSysMem = m_indices.toArray().getPtr();
232  hr = m_pDevice->CreateBuffer(&m_indexDesc, &initData, &m_pIndexBuffer);
233  if(FAILED(hr))
234  throw Exception("Failed to create vertex index buffer");
235 
236  m_areIndicesUploaded = true;
237  m_hasIndexBuffer = true;
238  m_uploadedIndices = m_indices.length();
239  }
240  else
241  {
242  D3D11_MAPPED_SUBRESOURCE res;
243  HRESULT hr = pDeviceContext->Map(m_pIndexBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &res);
244  GEP_ASSERT(SUCCEEDED(hr));
245  auto newData = m_indices.toArray();
246  memcpy(res.pData, newData.getPtr(), newData.length() * sizeof(uint32));
247  pDeviceContext->Unmap(m_pIndexBuffer, 0);
248  }
249  }
250  else
251  {
252  m_hasIndexBuffer = false;
253  }

257 void gep::Vertexbuffer::use(ID3D11DeviceContext* pDeviceContext)
258 {
259  UINT stride = m_elementSize;
260  UINT offset = 0;
261  pDeviceContext->IASetVertexBuffers(0, 1, &m_pDataBuffer, &stride, &offset);
262  if(m_hasIndexBuffer)
263  pDeviceContext->IASetIndexBuffer(m_pIndexBuffer, DXGI_FORMAT_R32_UINT, 0);
264  switch(m_primitive)
265  {
266  case Primitive::Triangle:
267  pDeviceContext->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
268  break;
269  case Primitive::Line:
270  pDeviceContext->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_LINELIST);
271  break;
272  default:
273  GEP_ASSERT(false, "unhandeled value");
274  }
