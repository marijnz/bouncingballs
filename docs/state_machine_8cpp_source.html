101 void gpp::sm::StateMachine::update(UpdateEventData* pData)
102 {
103  GEP_ASSERT(m_pCurrentState);
104  pData->m_pCurrentState = this;
105 
106  UpdateEventData updateData;
107  updateData.m_elapsedTime = pData->m_elapsedTime;
108  m_pCurrentState->update(&updateData);
109 
110  auto updateStepBehavior = updateData.getUpdateStepBehavior();
111  ConditionCheckResult conditionResult;
112 
113  if(updateStepBehavior != UpdateStepBehavior::LeaveWithNoConditionChecks)
114  {
115  conditionResult = m_pCurrentState->checkConditions();
116  }
117 
118  // If there are no next states to get into and we are supposed to continue with the current state, abort.
119  if(!conditionResult.hasNextState() && updateStepBehavior == UpdateStepBehavior::Continue)
120  {
121  return;
122  }
123 
124  // Leave the current state and enter the next one
126  auto pLeftState = m_pCurrentState;
127  LeaveEventData leaveData;
128  leaveData.m_pCurrentStateMachine = this;
129  leaveData.m_pNextState = conditionResult.nextState;
130  pLeftState->leave(&leaveData);
131 
132 
133  // Try to get the next state
134  GEP_ASSERT(leaveData.getNextState(), "There must be a next state after leaving the current state!");
135  m_pCurrentState = leaveData.getNextState();
136 
137  // If we can leave, signalize it to the parent state machine.
138  if(m_pCurrentState == m_pLeaveState)
139  {
140  pData->setUpdateStepBehavior(UpdateStepBehavior::Leave);
141  return;
142  }
143 
144  // Enter the next state
145  EnterEventData enterData;
146  enterData.m_pLeftState = pLeftState;
147  m_pCurrentState->enter(&enterData);
148 }

150 void gpp::sm::StateMachine::hookIntoUpdateFramework(gep::IUpdateFramework* ufx)
151 {
152  GEP_ASSERT(ufx, "Update framework is a nullptr.");
153  // register update()
154  auto id = ufx->registerUpdateCallback([=](float dt){
155  UpdateEventData data;
156  data.m_elapsedTime = dt;
157  update(&data);
158  if(data.getUpdateStepBehavior() != UpdateStepBehavior::Continue)
159  {
160  LeaveEventData data;

163  });
