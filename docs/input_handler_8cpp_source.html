58 void gep::XInputGamepad::update(float elapsedTime, DWORD dwUserIndex)
59 {
60  m_dwUserIndex = dwUserIndex;
61  if (m_checkStateTimer <= 0.f)
62  {
63  ZeroMemory(&m_XInputState, sizeof(XINPUT_STATE));
64  DWORD dwResult = XInputGetState(dwUserIndex, &m_XInputState);
65  if (dwResult == ERROR_SUCCESS)
66  {
67  XINPUT_GAMEPAD& gamepad = m_XInputState.Gamepad;
68 
69  // digital buttons
70  m_buttonsPressedOld = m_buttonsPressed;
71  m_buttonsPressed = gamepad.wButtons;
72  m_buttonsTriggered = (m_buttonsPressedOld ^ m_buttonsPressed) & m_buttonsPressed;
73  m_buttonsReleased = (m_buttonsPressedOld ^ m_buttonsPressed) & m_buttonsPressedOld;
74 
75  // analog sticks
76  m_leftStick = clampStick(gamepad.sThumbLX, gamepad.sThumbLY, XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE, SHRT_MAX - 5);
77  m_rightStick = clampStick(gamepad.sThumbRX, gamepad.sThumbRY, XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE, SHRT_MAX - 5);
78 
79  // shoulder buttons
80  m_leftTrigger = clampTrigger(gamepad.bLeftTrigger, XINPUT_GAMEPAD_TRIGGER_THRESHOLD, UCHAR_MAX - 5);
81  m_rightTrigger = clampTrigger(gamepad.bRightTrigger, XINPUT_GAMEPAD_TRIGGER_THRESHOLD, UCHAR_MAX - 5);
82 
83  // motors
84  updateRumble(elapsedTime);
85 
86  m_checkStateTimer = 0.f;
87  }
88  else
89  {
90  m_checkStateTimer = 2.f; // wait 2 seconds
91  }
92  }
93  else
94  {
95  m_checkStateTimer -= elapsedTime;
96  }
97 }
