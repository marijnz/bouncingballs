126  inline const mat3_t<T> submat(const int i, const int j) const
127  {
128  mat3_t<T> mb(DO_NOT_INITIALIZE);
129  int di, dj, si, sj;
130  // loop through 3x3 submatrix
131  for( di = 0; di < 3; di ++ ) {
132  for( dj = 0; dj < 3; dj ++ ) {
133  // map 3x3 element (destination) to 4x4 element (source)
134  si = di + ( ( di >= i ) ? 1 : 0 );
135  sj = dj + ( ( dj >= j ) ? 1 : 0 );
136  // copy element
137  mb.data[di * 3 + dj] = data[si * 4 + sj];
138  }
139  }
140  return mb;
141  }

158  inline const mat4_t<T> inverse() const
159  {
160  mat4_t<T> mr(DO_NOT_INITIALIZE);
161  T mdet = this->det();
162  mat3_t<T> mtemp(DO_NOT_INITIALIZE);
163  int sign=0;
164  if ( mdet > -GetEpsilon<T>::value() && mdet < GetEpsilon<T>::value() )
165  return mat4::identity();
166  for (int i = 0; i < 4; i++ )
167  for (int j = 0; j < 4; j++ ){
168  sign = 1 - ( (i +j) % 2 ) * 2;
169  mtemp = this->submat(i, j);
170  mr.data[i+j*4] = ( mtemp.det() * sign ) / mdet;
171  }

283  {
284  mat4_t<T> res;

289  //res.data[0] = xScale;

291  res.data[1] = 0; res.data[2] = 0; res.data[3] = 0;

296  res.data[4] = 0; res.data[6] = 0; res.data[7] = 0;

301  res.data[11] = -1;
302  res.data[8] = 0; res.data[9] = 0;

304  // W-Achse -zn*zf/(zf-zn)
305  //res.data[14] = -pNear * pFar / (pFar - pNear);
306  res.data[14] = (pFar * pNear) / (pNear - pFar);
307  res.data[12] = 0; res.data[13] = 0; res.data[15] = 0;
