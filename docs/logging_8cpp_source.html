168 void gep::ConsoleLogSink::take(LogChannel channel, const char* msg)
169 {
170  const char* channelName = "Unkown: ";
171  WORD color = 0x07; //white
172  switch(channel)
173  {
174  case LogChannel::message:
175  channelName = "";
176  break;
177  case LogChannel::warning:
178  channelName = "Warning: ";
179  color = 0x0E; //yellow
180  break;
181  case LogChannel::error:
182  color = 0x0C; // red
183  channelName = "Error: ";
184  break;
185  default:
186  GEP_ASSERT(false, "unhandeled log channel value");
187  break;
188  }
189  if(color != 0x07)
190  SetConsoleTextAttribute (GetStdHandle (STD_OUTPUT_HANDLE), color);
191  printf("%s%s\n", channelName, msg);
192  if(color != 0x07)
193  SetConsoleTextAttribute (GetStdHandle (STD_OUTPUT_HANDLE), 0x07); //back to white
194 }
