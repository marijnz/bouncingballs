107 void gpp::CameraComponent::setState(State::Enum state)
108 {
109  GEP_ASSERT(state != State::Initial, "Cannot set the initial state!");
110 
111  m_state = state;
112 
113  switch (state)
114  {
115  case State::Active:
116  g_globalManager.getCameraManager()->setActiveCamera(m_pCamera);
117  g_gameObjectManager.setCurrentCameraObject(m_pParentGameObject);
118  break;
119  case State::Inactive:
120  GEP_ASSERT(false, "Cannot directly set a camera to be Inactive."
121  "If you set any other camera to Active, all other cameras will "
122  "automatically be set to be Inactive.");
123  break;
124  default:
125  GEP_ASSERT(false, "Invalid state input!", state);
126  break;
127  }
128 }
