52 void gep::Model::doFindMinMax(mat4 transformation, const ModelLoader::NodeDrawData* pNode, vec3& minOut, vec3& maxOut)
53 {
54  GEP_ASSERT(pNode != nullptr, "pNode may not be null");
55  transformation = transformation * pNode->transform;
56  //writefln("pNode = %x",cast(void*)pNode);
57  //Search min-max for all meshes inside
58  for(auto meshIndex : pNode->meshes){
59  const ModelLoader::MeshData& meshData = m_modelLoader.getModelData().meshes[meshIndex];
60  vec3 min = transformation.transformPosition(meshData.bbox.getMin());
61  vec3 max = transformation.transformPosition(meshData.bbox.getMax());
62 
63  //The transformation may swapped a axis, fix if neccesary
64  if(min.x > max.x) std::swap(min.x,max.x);
65  if(min.y > max.y) std::swap(min.y,max.y);
66  if(min.z > max.z) std::swap(min.z,max.z);
67 
68  if(min.x < minOut.x) minOut.x = min.x;
69  if(min.y < minOut.y) minOut.y = min.y;
70  if(min.z < minOut.z) minOut.z = min.z;
71 
72  if(max.x > maxOut.x) maxOut.x = max.x;
73  if(max.y > maxOut.y) maxOut.y = max.y;
74  if(max.z > maxOut.z) maxOut.z = max.z;
75  }
76 
77  //Follow tree
78  for(auto& c : pNode->children){
79  doFindMinMax(transformation, c, minOut, maxOut);
80  }
81 }

141 void gep::Model::generateMeshes()
142 {
143  // Check if the model has been loaded correctly
144  GEP_ASSERT(m_modelLoader.getModelData().hasData, "No file has been loaded yet");
145 #ifdef _DEBUG
146  {
147  int i=0;
148  for(auto& mat : m_materials){
149  if(mat.getShader() == nullptr)
150  {
151  GEP_ASSERT(false, "no shader has been set for material", i);
152  }
153  i++;
154  }
155  }
156 #endif
157 
158  //Generate Meshes
159  m_meshDrawData.resize(m_modelLoader.getModelData().meshes.length());
160  size_t i = 0;
161  for(auto& mesh : m_modelLoader.getModelData().meshes)
162  {
163 
164  // Generate the array of attributes we can use
165  DynamicArray<Vertexbuffer::DataChannel> neededChannels;
166  neededChannels.reserve(5);
167 
168  neededChannels.append(Vertexbuffer::DataChannel::POSITION);
169  if(mesh.normals.length() > 0)
170  {
171  neededChannels.append(Vertexbuffer::DataChannel::NORMAL);
172  }
173  if(mesh.tangents.length() > 0)
174  {
175  neededChannels.append(Vertexbuffer::DataChannel::TANGENT);
176  }
177  if(mesh.bitangents.length() > 0)
178  {
179  neededChannels.append(Vertexbuffer::DataChannel::BINORMAL);
180  }
181  if(mesh.texcoords[0].length() > 0)
182  {
183  neededChannels.append(Vertexbuffer::DataChannel::TEXCOORD0);
184  }
185  if(mesh.boneInfos.length() > 0)
186  {
187  neededChannels.append(Vertexbuffer::DataChannel::BONE_INDICES);
188  neededChannels.append(Vertexbuffer::DataChannel::BONE_WEIGHTS);
189  }
190 
191  //Lets check if the mesh that we are loading has all the requested data channels
192  int VertexSize = 0;
193  for( auto channel : neededChannels)
194  {
195  switch(channel){
196  case Vertexbuffer::DataChannel::NORMAL:
197  if(mesh.normals.length() == 0){
198  std::ostringstream msg;
199  msg << "Error loading file '" << m_modelLoader.getFilename() << "' normal requested but not in mesh number " << i;
200  throw LoadingError(msg.str());
201  }
202  break;
203  case Vertexbuffer::DataChannel::BINORMAL:
204  case Vertexbuffer::DataChannel::TANGENT:
205  if(mesh.tangents.length() == 0 || mesh.bitangents.length() == 0){
206  std::ostringstream msg;
207  msg << "Error loading file '" << m_modelLoader.getFilename() << "' tangent and binormal requested but not in mesh number " << i;
208  throw LoadingError(msg.str());
209  }
210  break;
211  case Vertexbuffer::DataChannel::TEXCOORD0:
212  if(mesh.texcoords[(uint32)channel - (uint32)(Vertexbuffer::DataChannel::TEXCOORD0)].length() == 0){
213  std::ostringstream msg;
214  msg << "Error loading file '" << m_modelLoader.getFilename() << "' texcoord " << ((uint32)channel - (uint32)(Vertexbuffer::DataChannel::TEXCOORD0))
215  << " requested but not found in mesh number " << i;
216  throw LoadingError(msg.str());
217  }
218  break;
219  case Vertexbuffer::DataChannel::POSITION:
220  if(mesh.vertices.length() == 0){
221  std::ostringstream msg;
222  msg << "Error loading file '" << m_modelLoader.getFilename() << "' position requested but not in mesh number " << i;
223  throw LoadingError(msg.str());
224  }
225  break;
226  case Vertexbuffer::DataChannel::BONE_INDICES:
227  case Vertexbuffer::DataChannel::BONE_WEIGHTS:
228  if(mesh.boneInfos.length() == 0){
229  std::ostringstream msg;
230  msg << "Error loading file '" << m_modelLoader.getFilename() << "' bones requested but there are no bones in mesh number " << i;
231  throw LoadingError(msg.str());
232  }
233  break;
234  default:
235  {
236  std::ostringstream msg;
237  msg << "Error loading '" << m_modelLoader.getFilename() << "' request of not supported data channel (not implemented yet)";
238  throw LoadingError(msg.str());
239  }
240  }
241  }
242 
243  //Insert the data into the vertex buffer
244  Vertexbuffer* vb = m_pVertexbuffer;
245 
246  if(vb == nullptr){
247  vb = new Vertexbuffer(m_pDevice, neededChannels.toArray(), Vertexbuffer::Primitive::Triangle, Vertexbuffer::Usage::Static);
248  m_pVertexbuffer = vb;
249  }
250 
251  m_meshDrawData[i].vertexbuffer = vb;
252  m_meshDrawData[i].startIndex = (uint32)vb->getIndices().length();
253  m_meshDrawData[i].numIndices = (uint32)(mesh.faces.length() * 3);
254  m_meshDrawData[i].materialIndex = mesh.materialIndex;
255 
256  size_t IndexOffset = vb->getCurrentNumVertices();
257 
258  for(size_t j=0; j<mesh.vertices.length(); j++)
259  {
260  for(auto channel : neededChannels)
261  {
262  uint32 dataChannelSize = Vertexbuffer::getDataChannelSize(channel);
263  switch(channel){
264  case Vertexbuffer::DataChannel::POSITION:
265  GEP_ASSERT(dataChannelSize <= 3 && dataChannelSize > 0);
266  vb->getData().append(ArrayPtr<float>((float*)mesh.vertices[j].data, dataChannelSize));
267  break;
268  case Vertexbuffer::DataChannel::NORMAL:
269  GEP_ASSERT(dataChannelSize <= 3 && dataChannelSize > 0);
270  vb->getData().append(ArrayPtr<float>((float*)mesh.normals[j].data, dataChannelSize));
271  break;
272  case Vertexbuffer::DataChannel::BINORMAL:
273  GEP_ASSERT(dataChannelSize <= 3 && dataChannelSize > 0);
274  vb->getData().append(ArrayPtr<float>((float*)mesh.bitangents[j].data, dataChannelSize));
275  break;
276  case Vertexbuffer::DataChannel::TANGENT:
277  GEP_ASSERT(dataChannelSize <= 3 && dataChannelSize > 0);
278  vb->getData().append(ArrayPtr<float>((float*)mesh.tangents[j].data, dataChannelSize));
279  break;
280  case Vertexbuffer::DataChannel::TEXCOORD0:
281  {
282  int index = (int)channel - (int)Vertexbuffer::DataChannel::TEXCOORD0;
283  GEP_ASSERT(index >= 0 && index < 4);
284  GEP_ASSERT(dataChannelSize == 2); //currently only 2 component uv channels supported
285  GEP_ASSERT(mesh.texcoords[index].length() > 0);
286  vb->getData().append(ArrayPtr<float>((float*)mesh.texcoords[index][j].data, dataChannelSize));
287  }
288  break;
289  case Vertexbuffer::DataChannel::BONE_INDICES:
290  {
291  auto boneIndexData = ArrayPtr<float>((float*)mesh.boneInfos[j].boneIds, dataChannelSize);
292  vb->getData().append(boneIndexData);
293  }
294  break;
295  case Vertexbuffer::DataChannel::BONE_WEIGHTS:
296  {
297  auto boneWeightData = ArrayPtr<float>((float*)mesh.boneInfos[j].weights, dataChannelSize);
298  vb->getData().append(boneWeightData);
299  }
300  break;
301  default:
302  GEP_ASSERT(0, "Trying to add unsupported data type to vertex buffer");
303  }
304  }
305  }
306 
307  for(auto& face : mesh.faces){
308  uint32 indices[3];
309  for(int c=0;c<3;c++)
310  indices[c] = (uint32)(face.indices[c] + IndexOffset);
311  vb->getIndices().append(indices);
312  }
313  i++;
314  }
315 }

443 // Use this method if you want to create the combined bone matrix by yourself.
444 // Caution: You have to use bone transformations in local space, instead of modelspace, so
445 // in model.cpp use "m_pPose->getBoneLocalSpace(index)"
446 gep::mat4 gep::Model::accumulatedOffsetMatrix(const ArrayPtr<mat4>& transformations, const gep::ModelLoader::BoneNode& bone)
447 {
448  mat4 result = mat4::identity();
449  auto parentNode = bone.node->parent;
450  while(parentNode != nullptr)
451  {
452  if (parentNode->bone != nullptr)
453  {
454  auto parentBoneId = parentNode->bone - m_modelLoader.getModelData().bones.getPtr();
455  result = result * transformations[parentBoneId];
456  }
457  else
458  {
459  result = result * parentNode->transform;
460  }
461  parentNode = parentNode->parent;
462  }
463  return result;
464 }
