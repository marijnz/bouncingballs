103 gep::Result gep::TaskWorker::stealTasks()
104 {
105  // try to steal tasks from all other task workers but ourself
106  for(TaskWorker* pOtherWorker : m_pTaskQueue->m_worker)
107  {
108  if(pOtherWorker != this)
109  {
110  ScopedLock<Mutex> lock(pOtherWorker->m_tasksMutex);
111  if(pOtherWorker->m_tasks.length() > 0)
112  {
113  // steal half of the tasks, but at least 1
114  size_t tasksToSteal = pOtherWorker->m_tasks.length() / 2;
115  if(tasksToSteal < 1)
116  tasksToSteal = 1;
117  auto otherTasks = pOtherWorker->m_tasks.toArray();
118  size_t otherNewLength = otherTasks.length() - tasksToSteal;
119  // Append a sub array to our task queue
120  m_tasks.append( otherTasks(otherNewLength, otherTasks.length()) );
121  // remove the stolen tasks from the other task queue
122  pOtherWorker->m_tasks.resize(otherNewLength);
123  return SUCCESS;
124  }
125  }
126  }
127  return FAILURE;
128 }

213 void gep::TaskQueue::scheduleNextGroup()
214 {
215  ScopedLock<Mutex> lock(m_schedulingMutex);
216  if(m_remainingTaskGroups.count() > 0)
217  {
218  m_currentTaskGroup = m_remainingTaskGroups.take();
219  size_t numTasks = m_currentTaskGroup->m_tasks.length();
220  size_t tasksPerWorker = numTasks / m_worker.length();
221  if(tasksPerWorker < 1)
222  tasksPerWorker = 1;
223  size_t taskStart = 0;
224  auto taskArray = m_currentTaskGroup->m_tasks.toArray();
225  // set the active task group on all workers
226  for(auto pWorker : m_worker)
227  {
228  pWorker->m_pActiveGroup = m_currentTaskGroup;
229  }
230  // distribute the tasks to the workers
231  for(auto pWorker : m_worker)
232  {
233  if(taskStart >= numTasks)
234  break;
235  size_t end = taskStart + tasksPerWorker;
236  if(end > numTasks)
237  end = numTasks;
238  pWorker->addTasks( taskArray(taskStart, end) );
239  taskStart += tasksPerWorker;
240  }
241 
242  // wakeup all the workers but the first (the first is the local worker)
243  for(size_t i=1; i < m_worker.length(); i++)
244  m_worker[i]->m_hasWorkSemaphore.increment();
245  }
246  else
247  {
248  m_currentTaskGroup = nullptr;
249  }
250 }
