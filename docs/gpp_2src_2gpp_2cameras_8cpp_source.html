20 {
21  if(m_orthographic)
22  {
23  float distance = (m_far - m_near) / 8; // use 1/8 view distance as adjacent length to calculate frustum
24  float viewAngleHalfInRad = gep::toRadians(m_viewAngleInDegrees/2);
25  float tangens = gep::sin(viewAngleHalfInRad) / gep::cos(viewAngleHalfInRad);
26  float topDist = tangens * distance; // tan * adjacent = opposite
27  float rightDist = m_aspectRatio * topDist;
28  return gep::mat4::ortho(-topDist, topDist, -rightDist, rightDist, m_near, m_far);

43 gep::Ray gpp::Camera::getRayForNormalizedScreenPos(const gep::vec2& screenPos)
44 {
45  using namespace gep;
46  vec3 from;
47  vec3 to;
48  Ray result = Ray(from, to);
49 
50  if (isOrthographic())
51  {
52  float distance = (m_far - m_near) / 8; // use 1/8 view distance as adjacent length to calculate frustum
53  float viewAngleHalfInRad = gep::toRadians(m_viewAngleInDegrees/2);
54  float tangens = gep::sin(viewAngleHalfInRad) / gep::cos(viewAngleHalfInRad);
55  float topDist = tangens * distance; // tan * adjacent = opposite
56  float rightDist = m_aspectRatio * topDist;
57 
58  gep::vec3 offsetOnViewPlane = gep::vec3(0,0,0);
59  offsetOnViewPlane.x = screenPos.x * rightDist;
60  offsetOnViewPlane.y = screenPos.y * topDist; // screen coords have y upwards, but in gep, z is up direction
61 
62  from = getWorldPosition() + getRightDirection() * offsetOnViewPlane.x;
63  from = from + getUpDirection() * offsetOnViewPlane.y;
64 
65  result = Ray(from, getViewDirection());
66  }
67  else
68  {
69  mat4 inverseViewProjeciton = getProjectionMatrix() * getViewMatrix();
70  inverseViewProjeciton = inverseViewProjeciton.inverse();
71  vec4 temp = inverseViewProjeciton * vec4(screenPos.x, screenPos.y, 1, 1);
72  to = vec3(temp.x, temp.y, temp.z).normalized();
73 
74  result = Ray(getWorldPosition(), to);
75  }
76 
77  return result;
78 }
