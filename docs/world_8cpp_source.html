60 hkBool gep::HavokCollisionFilter_Simple::isCollisionEnabled(hkUint32 infoA,
61  hkUint32 infoB) const
62 {
63  // Examine the collisionfilterinfo of each. This is "user" information which can be interpreted
64  // however the filter sees fit. Another example would be to let every body have a unique
65  // collisionfilterinfo value, and allow/disallow collisions based on a pairwise lookup.
66  // We use a much simpler rule here for ease of illustration.
67  // Let's say that bodies with collisionfilters X and Y are allowed to collide if and only if X == Y,
68  // ie. if they are in the same "group"
69 
70  return infoA & infoB;
71 }

192 void gep::HavokWorld::castRay(const RayCastInput& input, RayCastOutput& output) const
193 {
194  hkpWorldRayCastInput actualInput;
195  hkpWorldRayCastOutput actualOutput;
196 
197  // Process input
198  conversion::hk::to(input.from, actualInput.m_from);
199  conversion::hk::to(input.to, actualInput.m_to);
200  actualInput.m_filterInfo = input.filterInfo;
201 
202  // Cast the ray
203  {
204  m_pWorld->lock();
205  m_pWorld->castRay(actualInput, actualOutput);
206  m_pWorld->unlock();
207  }
208 
209  // Process output
210  output.hitFraction = actualOutput.m_hitFraction;
211  conversion::hk::from(actualOutput.m_normal, output.normal);
212  if (actualOutput.m_rootCollidable)
213  {
214  // We only support rigid bodies right now.
215  auto actualEntity = static_cast<hkpEntity*>(actualOutput.m_rootCollidable->getOwner());
216  output.pHitBody = static_cast<IRigidBody*>(reinterpret_cast<IPhysicsEntity*>(actualEntity->getUserData()));
217  }
