136 void gpp::GameObject::initialize()
137 {
138  auto pAllocator = GameObjectManager::instance().getTempAllocator();
139 
140  auto toInit = gep::DynamicArray<ComponentWrapper*>(pAllocator);
141  toInit.reserve(m_components.count());
142 
143  for(auto& wrapper : m_components.values())
144  {
145  toInit.append(&wrapper);
146  }
147 
148  std::sort(toInit.begin(), toInit.end(), [](ComponentWrapper* lhs, ComponentWrapper* rhs){
149  return lhs->initializationPriority < rhs->initializationPriority;
150  });
151 
152  for(auto wrapper : toInit)
153  {
154  wrapper->component->initalize();
155  GEP_ASSERT(wrapper->component->getState() != IComponent::State::Initial,
156  "A game component must set its state within its initialize function!");
157  }
158 }

160 void gpp::GameObject::destroy()
161 {
162  auto pAllocator = GameObjectManager::instance().getTempAllocator();
163 
164  // Create a sorted array of all component instances.
165  auto toDestroy = gep::DynamicArray<ComponentWrapper*>(pAllocator);
166  toDestroy.reserve(m_components.count());
167 
168  for(auto& wrapper : m_components.values())
169  {
170  toDestroy.append(&wrapper);
171  }
172 
173  // sort so least prioritized components come first.
174  std::sort(toDestroy.begin(), toDestroy.end(), [](ComponentWrapper* lhs, ComponentWrapper* rhs){
175  return lhs->initializationPriority > rhs->initializationPriority;
176  });
177 
178  // Call destroy on all components
179  for(auto wrapper : toDestroy)
180  {
181  wrapper->component->destroy();
182 
183  }
184 
185  // Delete the components.
186  for(auto wrapper : toDestroy)
187  {
188  gep::deleteAndNull(wrapper->component);
189  }
190 
191  m_components.clear();
192  m_updateQueue.resize(0);
193 }
