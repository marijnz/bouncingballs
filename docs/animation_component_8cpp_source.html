41 void gpp::AnimationComponent::initalize()
42 {
43  GEP_ASSERT(m_skeletonPath != nullptr, "Can't initialize animation component without skeleton path", m_pParentGameObject->getName().c_str());
44  GEP_ASSERT(m_animationPaths.count() >= 1, "Can't initialize animation component without animation path", m_pParentGameObject->getName().c_str());
45  GEP_ASSERT(m_skinPath != nullptr, "Can't initialize animation component without skin path", m_pParentGameObject->getName().c_str());
46 
47  auto animationFactory = g_globalManager.getAnimationSystem()->getAnimationFactory();
48 
49  m_skeleton = animationFactory->loadAnimation(m_skeletonPath);
50  m_skins = animationFactory->loadAnimation(m_skinPath);
51 
52  m_AnimatedSkeleton = animationFactory->createAnimatedSkeleton(m_skeleton);
53 
54 
55  for(auto kv : m_animationPaths)
56  {
57  auto anim = animationFactory->loadAnimation(kv.value.c_str());
58  m_animations[kv.key] = anim;
59  m_animationControls[kv.key] = m_AnimatedSkeleton->addAnimationControl(anim);
60  }
61 
62  m_AnimatedSkeleton->setParentTransform(this->getParentGameObject());
63  RenderComponent* rc = m_pParentGameObject->getComponent<RenderComponent>();
64  if (rc != nullptr)
65  {
66  gep::DynamicArray<const char*> boneNames;
67  rc->getBoneNames(boneNames);
68  gep::DynamicArray<size_t> boneIds;
69  getBoneMapping(boneNames.toArray(), boneIds);
70  rc->setBoneMapping(boneIds);
71  }
72  setState(State::Active);
73 }

104 void gpp::AnimationComponent::setState( State::Enum state )
105 {
106  GEP_ASSERT(state != State::Initial);
107 
108  auto oldState = m_state;
109 
110  if (oldState == state) { return; } // Nothing to do here.
111 
112  switch (state)
113  {
114  case State::Active:
115  if (oldState != State::Active)
116  {
117  activate();
118  }
119  break;
120  case State::Inactive:
121  if (oldState == State::Active)
122  {
123  deactivate();
124  }
125  break;
126  default:
127  GEP_ASSERT(false, "Invalid requested state.", oldState, state);
128  break;
129  }
130 
131  m_state = state;
132 }
