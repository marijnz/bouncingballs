398  {
399  if (m_listeners.length() == 0)

403  }
404 
405  // reverse iteration
406  for (size_t index = m_listeners.length(); index > 0; --index)
407  {
408  auto& listener = m_listeners[index - 1];
409  if (listener.priority <= wrapper.priority)
410  {
411  m_listeners.insertAtIndex(index, wrapper);
412  return;
413  }
414  }
415  m_listeners.insertAtIndex(0, wrapper);
