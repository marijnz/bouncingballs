204 void gpp::PhysicsComponent::setState(State::Enum newState)
205 {
206  GEP_ASSERT(newState != State::Initial);
207 
208  auto oldState = m_state;
209 
210  if (oldState == newState) { return; } // Nothing to do here.
211 
212  switch (newState)
213  {
214  case State::Active:
215  if (oldState != State::Active)
216  {
217  activate();
218  }
219  break;
220  case State::Inactive:
221  if (oldState == State::Active)
222  {
223  deactivate();
224  }
225  break;
226  default:
227  GEP_ASSERT(false, "Invalid requested state.", oldState, newState);
228  break;
229  }
230 
231  m_state = newState;
232 }
