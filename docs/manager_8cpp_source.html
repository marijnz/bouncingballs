34 void gep::HavokPhysicsManager::initialize()
35 {
36  m_pFactory = new HavokPhysicsFactory(&m_factoryAllocator);
37  m_pFactory->initialize();
38 
39  // set up display manager
40  m_pDisplayManager = new HavokDisplayManager();
41  m_pDisplayManager->initialize();
42  if (m_options & Options::DebugDrawing)
43  m_pDisplayManager->setEnabled(true);
44 
45  // set up contexts
46  m_pPhysicsContext = new hkpPhysicsContext();
47 
48  // Make a list of the contexts, which is used later on.
49  hkArray<hkProcessContext*> contexts;
50  contexts.pushBack(m_pPhysicsContext);
51 
52  // set up processes
53 
54  hkArray<const char*> processNames;
55 
56  // Push back the default display so we can see the low level hkDebugDisplay lines
57  hkDebugDisplayProcess::registerProcess();
58  processNames.pushBack(hkDebugDisplayProcess::getName());
59 
60  hkpPhysicsContext::registerAllPhysicsProcesses();
61 
62  processNames.pushBack(hkpShapeDisplayViewer::getName());
63  processNames.pushBack(hkpBroadphaseViewer::getName());
64  processNames.pushBack(hkpActiveContactPointViewer::getName());
65  processNames.pushBack("Trigger Volumes");
66 
67  for (const char* processName : processNames)
68  {
69  hkProcess* process = hkProcessFactory::getInstance().createProcess(processName, contexts);
70  GEP_ASSERT(process, "Failed to create havok process - did you register all the processes for the product(s) you are using?");
71  process->m_inStream = nullptr;
72  process->m_outStream = nullptr;
73  process->m_processHandler = nullptr;
74  process->m_displayHandler = static_cast<hkDebugDisplayHandler*>(m_pDisplayManager);
75  process->init();
76  m_physicsProcesses.append(process);
77  }
78 
79  #ifdef GEP_USE_HK_VISUAL_DEBUGGER
80  g_globalManager.getLogging()->logMessage("setting up visual debugger");
81  // Setup the visual debugger
82  m_pVisualDebugger = new hkVisualDebugger(contexts);
83  m_pVisualDebugger->serve();
84  #endif // GEP_USE_HK_VISUAL_DEBUGGER
85 }
