64 gep::Texture2D* gep::Texture2DFileLoader::loadResource(Texture2D* pInPlace)
65 {
66  Texture2D* result = pInPlace;
67  bool isInPlace = true;
68  if(pInPlace == nullptr || pInPlace->isLoaded())
69  {
70  result = m_pRenderer->createTexture2D(m_filename.c_str(), this, TextureMode::Static);
71  isInPlace = false;
72  }
73  try {
74  DDSLoader loader(&g_stdAllocator);
75  loader.loadFile(m_filename.c_str());
76  if(loader.isCubemap())
77  {
78  if(!isInPlace)
79  deleteResource(result);
80  g_globalManager.getLogging()->logError("Error loading 2d texture from file '%s':\n The given file is a cubemap but is expected to be a 2d texture", m_filename.c_str());
81  return pInPlace;
82  }
83 
84  ImageFormat format;
85  ImageCompression compression;
86 
87  switch(loader.getDataFormat())
88  {
89  case DDSLoader::D3DFORMAT::R8G8B8:
90  format = ImageFormat::RGB8;
91  compression = ImageCompression::NONE;
92  break;
93  case DDSLoader::D3DFORMAT::A8R8G8B8:
94  format = ImageFormat::RGBA8;
95  compression = ImageCompression::NONE;
96  break;
97  case DDSLoader::D3DFORMAT::DXT1:
98  format = ImageFormat::COMPRESSED_RGB_DXT1;
99  compression = ImageCompression::PRECOMPRESSED;
100  break;
101  case DDSLoader::D3DFORMAT::DXT3:
102  format = ImageFormat::COMPRESSED_RGB_DXT1;
103  compression = ImageCompression::PRECOMPRESSED;
104  break;
105  case DDSLoader::D3DFORMAT::DXT5:
106  format = ImageFormat::COMPRESSED_RGB_DXT1;
107  compression = ImageCompression::PRECOMPRESSED;
108  break;
109  default:
110  {
111  if(!isInPlace)
112  deleteResource(result);
113  g_globalManager.getLogging()->logError("Error loading 2d texture from file '%s':\n The given textue format is not supported", m_filename.c_str());
114  return pInPlace;
115  }
116  }
117 
118  auto data = loader.getData();
119  result->getImageData().setData(data, data->imageData, data->width, data->height, format, compression);
120  result->setHasData(true);
121  return result;
122  }
123  catch(LoadingError& ex)
124  {
125  if(!isInPlace)
126  deleteResource(result);
127  g_globalManager.getLogging()->logError("Error loading 2d texture from file '%s':\n%s", m_filename.c_str(), ex.what());
128  return nullptr;
129  }
130  return pInPlace;
131 }

264 void gep::Texture2D::finalize()
265 {
266  if(m_mode == TextureMode::Dynamic &&
267  m_pTexture != nullptr &&
268  m_gpuWidth == m_data.getWidth() &&
269  m_gpuHeight == m_data.getHeight() &&
270  m_gpuFormat == m_data.getFormat())
271  {
272  // update already existing texture
273  D3D11_MAPPED_SUBRESOURCE res;
274  auto hr = m_pDeviceContext->Map(m_pTexture, 0, D3D11_MAP_WRITE_DISCARD, 0, &res);
275  GEP_ASSERT(SUCCEEDED(hr));
276  res.RowPitch = 0;
277  res.DepthPitch = 0;
278  auto newData = m_data.getData()[0];
279  memcpy(res.pData, newData.getPtr(), newData.length());
280  m_pDeviceContext->Unmap(m_pTexture, 0);
281  }
282  else
283  {
284  GEP_RELEASE_AND_NULL(m_pResourceView)
285  GEP_RELEASE_AND_NULL(m_pTexture)
286  // create a new texture
287  D3D11_TEXTURE2D_DESC desc;
288  desc.Width = (UINT)m_data.getWidth();
289  desc.Height = (UINT)m_data.getHeight();
290  desc.MipLevels = (UINT)m_data.getData().length();
291  desc.ArraySize = 1;
292  desc.Format = toDxgiFormat(m_data.getFormat());
293  desc.SampleDesc.Count = 1;
294  desc.SampleDesc.Quality = 0;
295  desc.Usage = (m_mode == TextureMode::Static) ? D3D11_USAGE_DEFAULT : D3D11_USAGE_DYNAMIC;
296  desc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
297  desc.CPUAccessFlags = (m_mode == TextureMode::Static) ? 0 : D3D11_CPU_ACCESS_WRITE;
298  desc.MiscFlags = 0;
299 
300  D3D11_SUBRESOURCE_DATA buffer[32];
301  memset(buffer, 0, sizeof(buffer));
302  GEP_ASSERT(m_data.getData().length() < GEP_ARRAY_SIZE(buffer));
303  uint32 i=0;
304  size_t mipWidth = m_data.getWidth();
305  for(auto& mipmapLevel : m_data.getData())
306  {
307  buffer[i].pSysMem = mipmapLevel.getPtr();
308  if(m_data.isCompressed())
309  {
310  if(m_data.getFormat() == ImageFormat::COMPRESSED_RGBA_DXT1 || m_data.getFormat() == ImageFormat::COMPRESSED_RGB_DXT1)
311  buffer[i].SysMemPitch = (UINT)(8 * GEP_MAX(mipWidth / 4, 1));
312  else
313  buffer[i].SysMemPitch = (UINT)(16 * GEP_MAX(mipWidth / 4, 1));
314  }
315  else
316  {
317  buffer[i].SysMemPitch = (UINT)(mipWidth * m_data.getSizeOfComponent() * m_data.getNumberOfComponents());
318  }
319  buffer[i].SysMemSlicePitch = 0;
320  i++;
321  mipWidth /= 2;
322  }
323 
324  HRESULT hr = m_pDevice->CreateTexture2D(&desc, buffer, &m_pTexture);
325  GEP_ASSERT(SUCCEEDED(hr), "failed to create 2d texture");
326 
327  D3D11_SHADER_RESOURCE_VIEW_DESC resDesc;
328  resDesc.Format = toDxgiFormat(m_data.getFormat());
329  resDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
330  resDesc.Texture2D.MostDetailedMip = 0;
331  resDesc.Texture2D.MipLevels = (UINT)m_data.getData().length();
332  hr = m_pDevice->CreateShaderResourceView(m_pTexture, &resDesc, &m_pResourceView);
333  GEP_ASSERT(SUCCEEDED(hr), "failed to create shader resource view");
334  }
335 }
