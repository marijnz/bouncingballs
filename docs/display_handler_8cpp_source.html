74  vec3 newTranslation(DO_NOT_INITIALIZE);
75  Quaternion newRotation(DO_NOT_INITIALIZE);
76  gep::conversion::hk::from(transform, newTranslation, newRotation);
77 
78  newRotation = newRotation * Quaternion(vec3(1,0,0), -90);
79  mat4 newTransform = mat4::translationMatrix(newTranslation) * newRotation.toMat4();
80  m_transformations[id] = newTransform;

87  auto pDataHolder = GEP_NEW(g_stdAllocator, HavokDataHolder)(geom, pGeom->m_triangles);
88  auto vertices = ArrayPtr<vec4>((vec4*)&pGeom->m_vertices[0], pGeom->m_vertices.getSize());
89  auto indices = pDataHolder->getIndices();
90 
91  DisplayGeometryInfo info;
92  info.pModel = g_globalManager.getRenderer()->loadModel(pDataHolder, vertices, indices);
93 
94  gep::conversion::hk::from(geom->getTransform(), newTranslation, newRotation);
95  newRotation = newRotation * Quaternion(vec3(1,0,0), -90);
96  info.transform = mat4::translationMatrix(newTranslation) * newRotation.toMat4();
97  a.append(info);
98  }

243  void HavokDisplayManager::setEnabled(bool value)
244  {
245  if (value)
246  {
247  if (m_extractionCallbackId.id != 0) return; // we are already enabled
248 
249  m_extractionCallbackId = g_globalManager.getRendererExtractor()->registerExtractionCallback(std::bind(&HavokDisplayManager::extract, this, std::placeholders::_1));
250  GEP_ASSERT(m_extractionCallbackId.id != 0, "Unable to register extraction!");
251  }
252  else
253  {
254  if (m_extractionCallbackId.id == 0) return; // we are already disabled
255 
256  g_globalManager.getRendererExtractor()->deregisterExtractionCallback(m_extractionCallbackId);
257  m_extractionCallbackId = CallbackId(0);
258  }
