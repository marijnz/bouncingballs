55  {
56  //Compute A, B and C coefficients
57  vec3_t<T> offset = ray.pos - pos;
58  T a = ray.dir.dot(ray.dir);
59  T b = 2 * ray.dir.dot(offset);
60  T c = offset.dot(offset) - radiusSquared;

62  //Find discriminant

64 
65  // if discriminant is negative there are no real roots, so return
66  // false as ray misses Sphere_t<T>

71
