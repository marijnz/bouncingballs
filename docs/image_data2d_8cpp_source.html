174 void gep::ImageData2D::insert(const ImageData2D& src, size_t x, size_t y, size_t mipmap)
175 {
176  GEP_ASSERT(src.m_format == m_format, "Can only insert image data of the same format");
177  GEP_ASSERT(x + src.m_width <= m_width, "Insert out of bounds on x axis");
178  GEP_ASSERT(y + src.m_height <= m_height, "Insert out of bounds on y axis");
179  GEP_ASSERT(m_data.length() > 0, "Image data to insert in has no data");
180  size_t sizeOfPixel = m_numComponents * m_sizeOfComponent;
181  size_t sizeOfRow = m_width * sizeOfPixel;
182 
183  auto dstData = m_data[mipmap];
184  auto srcData = src.getData()[mipmap];
185  size_t sizeOfSrcRow = src.m_width * sizeOfPixel;
186  for(size_t dstY=0; dstY<src.m_height; dstY++){
187  size_t dstIndex = (y + dstY) * sizeOfRow + x * sizeOfPixel;
188  size_t srcIndex = dstY * sizeOfSrcRow;
189  dstData(dstIndex, dstIndex + sizeOfSrcRow).copyFrom( srcData(srcIndex, srcIndex + sizeOfSrcRow) );
190  }
191 }

213 void gep::ImageData2D::loadFromFile(const char* filename, ImageCompression compression)
214 {
215  GEP_ASSERT(m_pAllocator != nullptr);
216  const size_t len = strlen(filename);
217  if(len > 4 && _stricmp(filename + len - 4, ".dds") == 0)
218  {
219  DDSLoader loader(m_pAllocator);
220 
221  loader.loadFile(filename);
222  if(loader.isCubemap())
223  {
224  std::ostringstream msg;
225  msg << "Trying to load file " << filename << " which is a cubemap as 2d texture";
226  throw Exception(msg.str());
227  }
228 
229  if(loader.getImages().length() > 1)
230  {
231  std::ostringstream msg;
232  msg << "Trying to loader file '" << filename << "' which is a texture array of size " << loader.getImages().length() << " as 2d texture";
233  throw Exception(msg.str());
234  }
235 
236  ImageFormat imgformat;
237 
238  switch(loader.getDataFormat())
239  {
240  case DDSLoader::D3DFORMAT::DXT1:
241  imgformat = ImageFormat::COMPRESSED_RGBA_DXT1;
242  GEP_ASSERT(compression != ImageCompression::NONE, "can not uncompress compressed dxt1 texture");
243  break;
244  case DDSLoader::D3DFORMAT::DXT3:
245  imgformat = ImageFormat::COMPRESSED_RGBA_DXT3;
246  GEP_ASSERT(compression != ImageCompression::NONE, "can not uncompress compressed dxt3 texture");
247  break;
248  case DDSLoader::D3DFORMAT::DXT5:
249  imgformat = ImageFormat::COMPRESSED_RGBA_DXT5;
250  GEP_ASSERT(compression != ImageCompression::NONE, "can not uncompress compressed dxt5 texture");
251  break;
252  default:
253  {
254  std::ostringstream msg;
255  msg << "trying to load file " << filename << " which has a unsupported format";
256  throw Exception(msg.str());
257  }
258  }
259 
260  setData(loader.getData(), loader.getImages()[0], loader.getWidth(), loader.getHeight(), imgformat, ImageCompression::PRECOMPRESSED);
261  }
262  else
263  {
264  std::ostringstream msg;
265  msg << "trying to load file '" << filename << "' which has a unkown format";
266  throw Exception(msg.str());
267  }
268 }
