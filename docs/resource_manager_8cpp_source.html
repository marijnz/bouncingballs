104 {
105  // strip-start "ResourceManagerFullImpl"
106  // strip-start "ResourceManagerTodo"
107  // strip-start "MultiThreading1"
108  {
109  ScopedLock<Mutex> lock(m_patchResourceMutex);
110  for(size_t i=0; i<m_resourcesToPatch.length();)
111  {
112  auto& info = m_resourcesToPatch[i];
113  IResource* pDummyResource = nullptr;
114  if(!m_resourceDummies.tryGet(std::string(info.pLoader->getResourceType()), pDummyResource))
115  {
116  GEP_ASSERT(false, "resource type not registered yet");
117  }
118  if(info.pResource == nullptr) // failed load
119  {
120  if(info.ptr == pDummyResource)
121  m_failedInitialLoad[info.pLoader] = true;
122 
123  info.pLoader->postLoad(info.ptr);
124  m_resourcesToPatch.removeAtIndexUnordered(i); //instead of i++
125  }
126  else if(info.isFinalized) // load successfull and finalized
127  {
128  if(m_failedInitialLoad.exists(info.pLoader))
129  m_failedInitialLoad.remove(info.pLoader);
130 
131  if(info.ptr == pDummyResource)
132  {
133  //replacing a dummy resource
134  invalidateAndReplace(info.ptr, info.pResource);
135  GEP_ASSERT(info.ptr == info.pResource);
136  }
137  else
138  {
139  // replacing a valid resource
140  if(info.ptr != info.pResource) // do not swap on inplace reloading
141  {
142  IResource* pResourceToReplace = info.ptr;
143  pResourceToReplace->swapPlaces(*info.pResource);
144  info.pLoader->deleteResource(pResourceToReplace);
145  GEP_ASSERT(info.ptr == info.pResource);
146  }
147  }
148 
149  info.pLoader->postLoad(info.ptr);
150  m_resourcesToPatch.removeAtIndexUnordered(i); //instead of i++
151  }
152  else
153  {
154  i++;
155  }
156  }
157  }
158 
159  // strip-end
160  m_timeSinceLastCheck -= elapsedTime;
161  if(m_timeSinceLastCheck <= 0.0f)
162  {
163  m_updateNum++;
164  ScopedLock<Mutex> lock(m_fileChangedLock);
165  m_dataDirWatcher.enumerateChanges([=](const char* filename, DirectoryWatcher::Action::Enum action)
166  {
167  if(action != DirectoryWatcher::Action::modified)
168  return;
169  std::string path = "data\\" + std::string(filename);
170  ReloadInfo info;
171  if(m_fileChangedListener.tryGet(path, info))
172  {
173  // Some modified events come in twice. Make sure to not handle them twice
174  if(info.updateNum == m_updateNum)
175  return;
176  g_globalManager.getLogging()->logMessage("Reloading '%s' resource from file '%s'.", info.pLoader->getResourceType(), filename);
177  m_fileChangedListener[path].updateNum = m_updateNum;
178  // the modified file might still be be open for writing, wait until its possible to read it
179  Sleep(10);
180  while(true)
181  {
182  RawFile file(path.c_str(), "r");
183  if(file.isOpen())
184  break;
185  Sleep(100);
186  }
187  try {
188  IResource* pDummyResource = nullptr;
189  IResource* pResourceToReload = info.pResource;
190  if(!m_resourceDummies.tryGet(std::string(info.pLoader->getResourceType()), pDummyResource))
191  {
192  GEP_ASSERT(false, "resource type not registered yet");
193  }
194  IResource* pNewResource = info.pLoader->loadResource((pDummyResource == pResourceToReload) ? nullptr : pResourceToReload);
195  if(pNewResource != nullptr)
196  {
197  if(m_failedInitialLoad.exists(info.pLoader))
198  m_failedInitialLoad.remove(info.pLoader);
199  if(pResourceToReload != nullptr && pResourceToReload != pDummyResource)
200  {
201  // swap if it did not reload inplace
202  if(pResourceToReload != pNewResource)
203  {
204  pNewResource->swapPlaces(*pResourceToReload);
205  removeFromNewList(pResourceToReload);
206  info.pLoader->deleteResource(pResourceToReload);
207  }
208  }
209  else
210  {
211  //replace the dummy resource
212  invalidateAndReplace(info.pResource, pNewResource);
213  }
214 
215  if(pNewResource->getFinalizeOptions() > 0)
216  {
217  // strip-start "MultiThreading1"
218  ScopedLock<Mutex> lock(m_newResourceMutex);
219  // strip-end
220  m_newResources.append(pNewResource);
221  }
222  pNewResource->setLoader(info.pLoader);
223  }
224  else
225  {
226  g_globalManager.getLogging()->logError("Failed to reload '%s' resource from file '%s'. Loader returned null.", info.pLoader->getResourceType(), filename);
227  }
228  }
229  catch(LoadingError& ex)
230  {
231  g_globalManager.getLogging()->logError("Failed to reload '%s' resource from file '%s' error:\n%s", info.pLoader->getResourceType(), filename, ex.what());
232  }
233  }
234  });
235  m_timeSinceLastCheck = 0.1f;
236  }
