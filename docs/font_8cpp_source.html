23  maxY = m_glyphs[0].m_top;
24  minY = m_glyphs[0].m_height - m_glyphs[0].m_top;
25  for(auto& g : m_glyphs)
26  {
27  maxWidth = (g.m_width > maxWidth) ? g.m_width : maxWidth;
28  maxY = (g.m_top > maxY) ? g.m_top : maxY;
29  minY = (g.m_top - g.m_height< minY) ? g.m_top - g.m_height: minY;
30  }
31  maxHeight = maxY - minY;
32  m_maxHeight = maxHeight;
33 
34  int size=64;
35  int x,y;
36  while(size < 2048){
37  x=y=0;
38  for(auto& g : m_glyphs)
39  {
40  if((x + g.m_width + 1) > size)
41  {
42  y += maxHeight;
43  x = 0;
44  }
45  x+=g.m_width + 1;
46  }
47  if(y > size)
48  size*=2;
49  else
50  break;
51  }
52  if(size == 2048){
53  throw Exception("Font texture is to large");
54  }
55 
56  texture.createEmpty(size, size, ImageFormat::R8);
57  ImageData2D& ImageData = texture.getImageData();
58  memset(ImageData.getData()[0].getPtr(), 0, ImageData.getData()[0].length());
59 
60  x=y=0;
61  for(auto& g : m_glyphs)
62  {
63  if((x + g.m_width + 1) > size){
64  y += maxHeight;
65  x=0;
66  }
67  if(g.m_data != nullptr)
68  {
69  ImageData.insert(*g.m_data, x, y + maxHeight - g.m_top + minY);
70  delete g.m_data;
71  g.m_data = nullptr;
72  }
73  float fStep = 1.0f / (float)size;
74  g.m_minTexY = (float)y * fStep;
75  g.m_minTexX = (float)(x) * fStep;
76  x += g.m_width;
77  g.m_maxTexY = (float)(y + maxHeight) * fStep;
78  g.m_maxTexX = (float)(x) * fStep;
79  x++;
80  g.m_fWidth = (float)(g.m_width + g.m_right);

86  FT_Face face = (FT_Face)pFace;
87  // Load the glyph
88  if(FT_Load_Glyph( face, FT_Get_Char_Index( face, c ), FT_LOAD_DEFAULT ) != 0)

90  std::ostringstream msg;

92  throw Exception(msg.str());
93  }
94 
95  FT_Glyph glyph;
96  FT_Get_Glyph( face->glyph, &glyph);
97 
98  // Convert it to a bitmap
99  FT_Glyph_To_Bitmap( &glyph, FT_RENDER_MODE_NORMAL, nullptr, 1 );
100  FT_BitmapGlyph bitmap_glyph((FT_BitmapGlyph)glyph);
101  FT_Bitmap* bitmap = &bitmap_glyph->bitmap;
102 
103  // Allocate the bitmap
104  ImageData2D::image_data_t buffer;
105  if(bitmap->width > 0 || bitmap->rows > 0) {
106  buffer = GEP_NEW_ARRAY(g_stdAllocator, ImageData2D::mipmap_data_t, 1);
107  buffer[0] = GEP_NEW_ARRAY(g_stdAllocator, ImageData2D::channel_data_t, bitmap->rows * bitmap->width);
108  auto& mipmap = buffer[0];
109  for(int i=0; i<bitmap->rows * bitmap->width; i++){
110  mipmap[i] = bitmap->buffer[i];
111  }
112  }
113 
114  auto& g = m_glyphs[num];
115  if(bitmap->width > 0 || bitmap->rows > 0) {
116  g.m_data = GEP_NEW(g_stdAllocator, ImageData2D);
117  g.m_data->setData(&g_stdAllocator, buffer, bitmap->width, bitmap->rows, ImageFormat::R8, ImageCompression::NONE);
118  }
119  else
120  {
121  g.m_data = nullptr;
122  }
123  g.m_width = bitmap->width;
124  g.m_height = bitmap->rows;
125  g.m_top = bitmap_glyph->top;
126  g.m_left = bitmap_glyph->left;
127  g.m_right = (face->glyph->advance.x >> 6) - bitmap_glyph->left - bitmap->width;

209 void gep::Font::print(Vertexbuffer& pFontBuffer, const char* text, FontHorizontalOrientation orientation /*= Orientation::Left*/) const
210 {
211  GEP_ASSERT(m_isPrintable, "font is not printable yet");
212  float TempBuffer[16];
213  float maxHeight = (float)m_maxHeight;
214  float offsetX=0.0f, offsetY = 0.0f;
215 
216  switch (orientation)
217  {
218  case gep::FontHorizontalOrientation::Left:
219  break;
220  case gep::FontHorizontalOrientation::Centered:
221  {
222  int width = 0;
223  int height = 0;
224  getTextSize(width, height, text);
225  offsetX = -width / 2.0f;
226  }
227  break;
228  default:
229  GEP_ASSERT(false, "Unsupported orientation type!");
230  break;
231  }
232 
233  size_t len = strlen(text);
234  for(size_t i=0; i<len; i++)
235  {
236  auto c = text[i];
237  GEP_ASSERT(c > 0, "ascii string contains non-ascii characters. Please use wide char version to print non ascii characters");
238  size_t index = 0;
239  if(c == '\n'){
240  offsetY += floor(maxHeight * 1.5f);
241  offsetX = 0.0;
242  continue;
243  }
244  if(c >= m_startIndex && c <= m_endIndex)
245  index = m_charAsignment[c - m_startIndex];
246 
247  //Print the glyph
248  auto& glyph = m_glyphs[index];
249  offsetX += (float)glyph.m_left;
250 
251  TempBuffer[0] = offsetX; TempBuffer[1] = offsetY;
252  TempBuffer[2] = glyph.m_minTexX; TempBuffer[3] = glyph.m_minTexY;
253 
254  TempBuffer[4] = offsetX; TempBuffer[5] = maxHeight+offsetY;
255  TempBuffer[6] = glyph.m_minTexX; TempBuffer[7] = glyph.m_maxTexY;
256 
257  TempBuffer[8] = glyph.m_realWidth+offsetX; TempBuffer[9] = maxHeight+offsetY;
258  TempBuffer[10] = glyph.m_maxTexX; TempBuffer[11] = glyph.m_maxTexY;
259 
260  TempBuffer[12] = glyph.m_realWidth+offsetX; TempBuffer[13] = offsetY;
261  TempBuffer[14] = glyph.m_maxTexX; TempBuffer[15] = glyph.m_minTexY;
262 
263  uint32 startIndex = (uint32)pFontBuffer.getCurrentNumVertices();
264  uint32 indices[] = { startIndex, startIndex + 2, startIndex + 1, //first triangle
265  startIndex , startIndex + 3, startIndex + 2 }; // second triangle
266  pFontBuffer.getIndices().append(indices);
267  pFontBuffer.getData().append(TempBuffer);
268 
269  offsetX += glyph.m_fWidth;
270  }
