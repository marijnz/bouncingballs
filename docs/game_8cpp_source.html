123 void gpp::Game::update(float elapsedTime)
124 {
125  if (!m_continueRunningGame)
126  {
127  return;
128  }
129 
130  auto pInputHandler = g_globalManager.getInputHandler();
131  auto pPhysicsSystem = g_globalManager.getPhysicsSystem();
132  auto pRenderer = g_globalManager.getRenderer();
133 
134  if (pInputHandler->wasTriggered(gep::Key::F9))
135  {
136  pPhysicsSystem->setDebugDrawingEnabled(!pPhysicsSystem->getDebugDrawingEnabled());
137  }
138 
139  if (pInputHandler->wasTriggered(gep::Key::F8)) // Toggle VSync
140  {
141  m_vsyncMode.advanceToNext();
142  updateVSyncState();
143  }
144 
145  if (pInputHandler->wasTriggered(gep::Key::F7))
146  {
147  toggleDisplayMemoryStatistics();
148  }
149 
150  if (pInputHandler->wasTriggered(gep::Key::F6))
151  {
152  m_printDebugInfo = !m_printDebugInfo;
153  }
154 
155  if (m_printDebugInfo)
156  {
157  auto averageElapsedTime = g_globalManager.getUpdateFramework()->calcElapsedTimeAverage(60);
158 
159  auto& debugRenderer = pRenderer->getDebugRenderer();
160  debugRenderer.printText(pRenderer->toNormalizedScreenPosition(uvec2(10, 10)), gep::format("Current FPS: %.1f", 1 / elapsedTime).c_str());
161  debugRenderer.printText(pRenderer->toNormalizedScreenPosition(uvec2(10, 30)), gep::format("Average FPS: %.1f (last 60 frames)", 1 / averageElapsedTime).c_str());
162  debugRenderer.printText(pRenderer->toNormalizedScreenPosition(uvec2(10, 50)), gep::format("VSync: %s", m_vsyncMode.toString()).c_str());
163  }
164 
165  g_gameObjectManager.update(elapsedTime);
166 
167  auto pSoundSystem = g_globalManager.getSoundSystem();
168  auto activeCamObj = g_gameObjectManager.getCurrentCameraObject();
169  if(activeCamObj)
170  {
171  pSoundSystem->setListenerPosition(activeCamObj->getWorldPosition());
172  pSoundSystem->setListenerOrientation(activeCamObj->getWorldRotation());
173  }
174  else
175  {
176  pSoundSystem->setListenerPosition(vec3());
177  pSoundSystem->setListenerOrientation(Quaternion());
178  }
179 }
