60 void gep::LeakDetector::findLeaks(std::ostream& printTo)
61 {
62  Hashmap<AllocationInfo*, LeakInfo, HashMethodPointerPolicy> resolvedCallstacks(&m_allocator);
63 
64  void* maxAddr = nullptr;
65  void* minAddr = (void*)((size_t)0 - 1);
66  size_t leakedMemory = 0;
67 
68  // Resolve each identical leak exactly once
69  for(auto& leak : m_allocationMap)
70  {
71  leak.value->referenced = false;
72  if(resolvedCallstacks.exists(leak.value))
73  {
74  resolvedCallstacks[leak.value].count++;
75  }
76  else
77  {
78  LeakInfo info = { nullptr, 1 };
79  if(leak.value->backtraceSize > 0)
80  {
81  info.resolvedFunctions = (char*)m_allocator.allocateMemory(LINE_LENGTH * leak.value->backtraceSize);
82  StackWalker::resolveCallstack(ArrayPtr<StackWalker::address_t>(leak.value->backtrace, leak.value->backtraceSize), info.resolvedFunctions, LINE_LENGTH);
83  }
84  resolvedCallstacks[leak.value] = info;
85  }
86 
87  if(leak.key < minAddr)
88  minAddr = leak.key;
89  if(leak.key > maxAddr)
90  maxAddr = leak.key;
91  leakedMemory += leak.value->size;
92  }
93 
94  // Find the root leaks
95  for(auto& leak : m_allocationMap)
96  {
97  ArrayPtr<void*> possibleReferences((void**)leak.key, leak.value->size / sizeof(void*));
98  for(void* possibleReference : possibleReferences)
99  {
100  if(possibleReference >= minAddr && possibleReference <= maxAddr)
101  {
102  if(m_allocationMap.exists(possibleReference))
103  m_allocationMap[possibleReference]->referenced = true;
104  }
105  }
106  }
107 
108  // First print all the memory leaks
109  for(auto& leak : resolvedCallstacks)
110  {
111  printTo << "------------------------------------------" << std::endl;
112  printTo << "Memory leak size " << leak.key->size << " bytes " << leak.value.count << " times" << std::endl;
113  if(leak.value.resolvedFunctions == nullptr)
114  {
115  printTo << "No callstack available" << std::endl;
116  }
117  else
118  {
119  char* line = leak.value.resolvedFunctions;
120  for(int i = 0; i<leak.key->backtraceSize; ++i, line += LINE_LENGTH)
121  {
122  printTo << line << std::endl;
123  }
124  }
125  printTo << std::endl;
126  }
127 
128  // Now print all the root leaks
129  printTo << "------------------------------------------" << std::endl;
130  printTo << "Root Leaks" << std::endl;
131  for(auto& leak : m_allocationMap)
132  {
133  if(!leak.value->referenced)
134  {
135  printTo << "------------------------------------------" << std::endl;
136  printTo << "Memory leak size " << leak.value->size << std::endl;
137  LeakInfo info = { nullptr, 0 };
138  resolvedCallstacks.tryGet(leak.value, info);
139  if(info.resolvedFunctions == nullptr)
140  {
141  printTo << "No callstack available" << std::endl;
142  }
143  else
144  {
145  char* line = info.resolvedFunctions;
146  for(int i = 0; i<leak.value->backtraceSize; ++i, line += LINE_LENGTH)
147  {
148  printTo << line << std::endl;
149  }
150  }
151  }
152  }
153 
154  //Print statistics
155  printTo << std::endl;
156  printTo << m_allocationMap.count() << " memory leaks found. " << leakedMemory << " bytes memory leaked." << std::endl;
157 }

202 {

204  auto numPages = (size - 1) / s_pageSize + 2;
205  auto sizeNeeded = numPages * s_pageSize;
206 
207  auto address = static_cast<char*>(VirtualAlloc(
208  nullptr, // optional starting address
209  sizeNeeded,
210  MEM_COMMIT | MEM_RESERVE,
211  PAGE_READWRITE
212  ));
213 
214  if (address == nullptr)
215  {
216  wchar_t buffer[2048];
217  DWORD errorCode = GetLastError();
218  FormatMessage(
219  FORMAT_MESSAGE_FROM_SYSTEM,
220  nullptr,
221  errorCode,
222  0,
223  (LPTSTR)buffer,
224  2048,
225  nullptr);
226  GEP_ASSERT(false, "Error allocating memory!");
227  }
228 
229 
230  auto padding = s_pageSize - (size % s_pageSize);
231  if (size % s_pageSize == 0)
232  {
233  padding = 0;
234  }
235  else
236  {
237  padding &= ~0x7;

239

242

248  &unused
249  );
