139 gep::CollisionMesh* gep::CollisionMeshFileLoader::loadResource(CollisionMesh* pInPlace)
140 {
141  CollisionMesh* result = nullptr;
142  bool isInPlace = true;
143  if (pInPlace == nullptr)
144  {
145  result = new CollisionMesh();
146  isInPlace = false;
147  }
148  auto* havokLoader = g_resourceManager.getHavokResourceLoader();
149  auto* container = havokLoader->load(m_path.c_str());
150  GEP_ASSERT(container != nullptr, "Could not load asset! %s", m_path.c_str());
151 
152  if (container)
153  {
154  auto* physicsData = reinterpret_cast<hkpPhysicsData*>(container->findObjectByType(hkpPhysicsDataClass.getName()));
155  GEP_ASSERT(physicsData != nullptr, "Unable to load physics data!");
156 
157  if (physicsData)
158  {
159  const auto& physicsSystems = physicsData->getPhysicsSystems();
160  GEP_ASSERT(physicsSystems.getSize() == 1, "Wrong number of physics systems!");
161  auto* body = physicsSystems[0]->getRigidBodies()[0];
162  const auto* hkShape = body->getCollidable()->getShape();
163 
164  auto shape = conversion::hk::from(const_cast<hkpShape*>(hkShape));
165 
166  auto type = shape->getShapeType();
167  if ( type == hkcdShapeType::TRIANGLE ||
168  type == hkcdShapeType::BV_COMPRESSED_MESH ||
169  type == hkcdShapeType::CONVEX_VERTICES )
170  {
171  const hkTransform& transform = body->getTransform();
172  Transform* tempTrans = new Transform();
173  conversion::hk::from(transform, *tempTrans);
174 
175  // Since havok content tools are buggy (?) and no custom transformation can be applied,
176  // we have to convert into our engine's space by hand.
177  // TODO: Ensure, that this transformation is correct in every case
178  tempTrans->setRotation(tempTrans->getRotation() * Quaternion(vec3(1,0,0),180));
179  static_cast<HavokMeshShape*>(shape)->setTransform(tempTrans);
180  }
181 
182 
183  result->setShape(shape);
184 
185  }
186 
187  }
188  return result;
189 }

211 void gep::HavokPhysicsFactoryAllocator::freeMemory(void* mem)
212 {
213  if (mem == nullptr)
214  {
215  return;
216  }
217 
218  Allocation allocation;
219 
220  if(m_allocations.tryGet(mem, allocation) == FAILURE)

224 
225  m_allocations.remove(mem);
